class Solution {
  public:
    string firstRepChar(string s) {
        unordered_set<char> seen;

        for (char c : s) {
            if (seen.count(c)) {
                return string(1, c);  // return first repeating character
            }
            seen.insert(c);
        }

        return "-1";  // No character repeats
    }
};
