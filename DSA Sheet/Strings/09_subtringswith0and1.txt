class Solution {
  public:
    int maxSubStr(string str) {
        int count0=0;
        int count1=0;
        int res=0;
        for(char ch : str){
            if(ch=='0') count0++;
            else count1++;
            
            if(count0==count1)res++;
            }
            if(count0!=count1)return -1;
        return res;
    }
};
