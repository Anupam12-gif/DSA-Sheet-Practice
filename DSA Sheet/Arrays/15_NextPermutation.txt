class Solution {
  public:
    void nextPermutation(vector<int>& arr) {
        int n=arr.size();
        int i=n-2;
        while(i>=0 && arr[i]>=arr[i+1]){
            i--;
        }
        if(i>=0){
        for(int j=n-1;j>i;j--){
            if(arr[j]>arr[i]){
                swap(arr[i],arr[j]);
                break;
            }
        }
        }
        reverse(arr.begin()+i+1,arr.end());
    }
};
