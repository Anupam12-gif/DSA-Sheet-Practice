class Solution {
  public:
    vector<int> findDuplicates(vector<int>& arr) {
        int n=arr.size();
        unordered_map<int,int> freq;
        vector<int> ans;
        for(int i=0;i<n;i++){
           freq[arr[i]]++; 
        }
        for(auto it:freq){
            if(it.second>1)
            ans.push_back(it.first);
        }
        if(ans.empty())
        return {};
        sort(ans.begin(),ans.end());
        return ans;
    }
};
