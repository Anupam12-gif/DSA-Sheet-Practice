// ðŸ’¡ Approach: Use unordered_set to store unique elements from both arrays

class Solution {
  public:
    int findUnion(vector<int>& a, vector<int>& b) {
        unordered_set<int> s;
        for(int i=0;i<a.size();i++){
            s.insert(a[i]);
        }
         for(int i=0;i<b.size();i++){
            s.insert(b[i]);
        }
        return s.size();
    }
};
